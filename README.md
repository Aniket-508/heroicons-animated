## Installation

### Using shadcn CLI

```bash
pnpm dlx shadcn add @heroicons-animated/heart
```

### Using npm packages

```bash
pnpm add @heroicons-animated/react motion
```

```tsx
import { BeakerIcon } from "@heroicons-animated/react";

export default function App() {
  return (
    <>
      <BeakerIcon className="size-6" />
    </>
  );
}
```
